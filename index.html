/** * MASTER BACKEND | FAST SYSTEM * Optimized for Bulk Download & Surgical Sync */ const SPREADSHEET_ID = "1V6BuVDf3CNkzsY3smOO9TC48-_TWucpPpMSuTXSe1mA"; const DATA_SHEET = "Xcelerate Data"; const AUTH_SHEET = "User"; function doGet(e) { const action = e.parameter.action; try { if (action === "login") return handleLogin(e.parameter.username, e.parameter.password); if (action === "sync_surgical") return handleSurgicalSync(e.parameter.hashes); return createResponse({result: "error", message: "Invalid action"}); } catch (err) { return createResponse({result: "error", message: err.toString()}); } } function handleLogin(u, p) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(AUTH_SHEET); const data = sheet.getDataRange().getValues(); const user = String(u || "").toLowerCase().trim(); const pass = String(p || "").trim(); const userRow = data.slice(1).find(r => String(r[0]).toLowerCase().trim() === user); if (!userRow) return createResponse({result: "denied", reason: "User not found"}); if (String(userRow[1]).trim() === pass) return createResponse({ result: "success" }); return createResponse({ result: "denied", reason: "Password mismatch" }); } function handleSurgicalSync(clientHashesString) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(DATA_SHEET); const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = data.slice(1); let clientHashes = {}; try { clientHashes = JSON.parse(clientHashesString || "{}"); } catch(e) { clientHashes = {}; } const fNumIdx = headers.indexOf("File Number"); const updates = []; rows.forEach((row, index) => { // Unique ID: File Number or Row Index const rowId = (fNumIdx !== -1 && row[fNumIdx]) ? String(row[fNumIdx]) : "row-" + (index + 2); const rowContent = row.join("|"); const serverHash = Utilities.base64Encode(Utilities.computeDigest(Utilities.DigestAlgorithm.MD5, rowContent)); // If client doesn't have this ID or the data has changed (hash mismatch), send it if (!clientHashes[rowId] || clientHashes[rowId] !== serverHash) { updates.push({ data: row, hash: serverHash, id: rowId }); } }); return createResponse({ result: "success", headers: headers, updates: updates }); } function createResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
